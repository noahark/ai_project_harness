#!/usr/bin/env python3
"""RC4 authorized-exception construction tests (TEMPORARY; not a delivery file).

Builds throwaway git repos + status.json and runs the REAL validator to prove
the 7 dispatch cases + zero regression + degenerate behavior. Validates by
calling scripts/validate-stage.py exactly as the harness does.
"""
import subprocess, json, os, tempfile, importlib.util
from pathlib import Path

REPO = "/Users/ark/Desktop/ai code/ai_project_harness"
VALIDATOR = os.path.join(REPO, "scripts", "validate-stage.py")
STAGE_REL = "reports/agent-runs/test-stage"
WAIVER = "reports/agent-runs/test-stage/waiver.md"

# import the validator module (filename has a hyphen) to recompute real fingerprints
spec = importlib.util.spec_from_file_location("validate_stage", VALIDATOR)
vs = importlib.util.module_from_spec(spec)
spec.loader.exec_module(vs)


def git(args, cwd):
    r = subprocess.run(["git"] + args, cwd=cwd, capture_output=True, text=True)
    if r.returncode != 0:
        raise RuntimeError(f"git {args} in {cwd}: {r.stderr.strip()}")
    return r.stdout.strip()


def make_repo(tmp, n_extra):
    """Return commit shas [c0, c1, ...]; c0 is base, last is head."""
    subprocess.run(["git", "init", "-q"], cwd=tmp, check=True)
    git(["config", "user.email", "t@t"], tmp)
    git(["config", "user.name", "t"], tmp)
    git(["config", "commit.gpgsign", "false"], tmp)
    f = tmp / "src.txt"
    f.write_text("base\n")
    git(["add", "."], tmp); git(["commit", "-q", "-m", "c0"], tmp)
    shas = [git(["rev-parse", "HEAD"], tmp)]
    for i in range(1, n_extra + 1):
        f.write_text(f.read_text() + f"line{i}\n")
        git(["add", "."], tmp); git(["commit", "-q", "-m", f"c{i}"], tmp)
        shas.append(git(["rev-parse", "HEAD"], tmp))
    return shas


def fp(root, stage_dir, base, head):
    return vs.compute_diff_fingerprint(root, stage_dir, base, head)


def write_and_commit(tmp, doc):
    stage_dir = tmp / STAGE_REL
    stage_dir.mkdir(parents=True, exist_ok=True)
    (stage_dir / "waiver.md").write_text("user review-waiver authorization (verbatim)\n")
    # validate_required_files demands these exist at pre-accept; they live in
    # the stage commit (after base..head), so they do not affect the fingerprint.
    for name in ["00-task.md", "10-design.md", "11-adr.md", "20-implementation.md",
                 "60-test-output.txt", "70-handoff.md", "30-review-1.md", "50-review-2.md"]:
        if not (stage_dir / name).exists():
            (stage_dir / name).write_text("placeholder\n")
    (stage_dir / "status.json").write_text(json.dumps(doc, indent=2))
    git(["add", "-A"], tmp); git(["commit", "-q", "-m", "stage"], tmp)


def run_validator(tmp):
    r = subprocess.run(
        ["python3", VALIDATOR, "test-stage", "--phase", "pre-accept"],
        cwd=tmp, capture_output=True, text=True,
    )
    return r.returncode, (r.stdout + r.stderr).rstrip()


def base_doc(base, head, status_fp, review1_fp=None, review2_fp=None):
    return {
        "stage_id": "test-stage", "status": "accepted", "workflow": "stage-delivery",
        "implementer": "claude_glm",
        "base_sha": base, "head_sha": head, "diff_fingerprint": status_fp,
        "tests": {"status": "pass", "commands": ["echo"]},
        "review_1": {"reviewer": "kimi", "provider": "kimi", "model": "kimi",
                     "verdict": "ACCEPT", "json_schema_valid": True,
                     "diff_fingerprint": review1_fp if review1_fp is not None else status_fp,
                     "reviewer_prior_involvement": "none"},
        "review_2": {"reviewer": "codex", "provider": "codex", "model": "gpt",
                     "primary_provider": "codex",
                     "verdict": "ACCEPT", "json_schema_valid": True,
                     "diff_fingerprint": review2_fp if review2_fp is not None else status_fp,
                     "reviewer_prior_involvement": "none"},
        "tasks": [],
    }


def exc(scope, applies, evidence=WAIVER, authorizer="user"):
    return {"assertion_id": "review_fingerprint_trails_status", "scope": scope,
            "applies_to_fingerprint": applies, "reason": "user review-waiver",
            "authorizer": authorizer, "evidence_file": evidence, "at": "2026-07-17T00:00:00Z"}


def new_tmp():
    return Path(tempfile.mkdtemp(prefix="rc4_"))

    doc = base_doc(base, head, sfp, review1_fp="STALE:1", review2_fp="STALE:2")
    doc["authorized_exceptions"] = [exc("review_1", sfp), exc("review_2", sfp)]
    write_and_commit(tmp, doc)
    return run_validator(tmp)

def main():

    for label, fn, expected_rc in CASES:

        print(out)
    print("\n===== SUMMARY =====")
    print("ALL CASES AS EXPECTED" if all_ok else "SOME CASES UNEXPECTED")


if __name__ == "__main__":
    main()
